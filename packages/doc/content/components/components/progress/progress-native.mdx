### Usage

```javascript type=expo
<ScrollView
  style={{
    width: '100%',
    height: '100%',
    padding: 80,
  }}
>
  <Progress max={100} value={35} />

  <Progress max={100} value={20} label={{ value: 20, placement: 'left' }} />
  <Progress max={100} value={40} label={{ value: 40, placement: 'left' }} />
  <Progress max={100} value={67} label={{ value: 67, placement: 'left' }} />

  <Progress
    max={100}
    value={70}
    label={{ value: 'With margin prop', placement: 'left' }}
    variant="secondary"
    mv="medium"
  />

  <Progress
    max={100}
    value={70}
    label={{ value: 'With variant', placement: 'left' }}
    variant="secondary"
  />
</ScrollView>
```
